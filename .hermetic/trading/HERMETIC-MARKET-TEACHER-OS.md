**Scoring System** (from your research):
```javascript
let bullishScore = 0, bearishScore = 0;

// RSI signals
if (rsi < 30) bullishScore += 2;
if (rsi > 70) bearishScore += 2;

// MACD signals
if (macd > signal) bullishScore += 2;
else bearishScore += 2;
if (histogram > 0) bullishScore += 1;
else bearishScore += 1;

// Bollinger Bands
if (price < lowerBand) bullishScore += 2;
if (price > upperBand) bearishScore += 2;

// Moving Averages
if (price > sma50 && sma50 > sma200) bullishScore += 2;
if (price < sma50 && sma50 < sma200) bearishScore += 2;

// Volume confirmation
if (volume > avgVolume * 1.5) {
  if (bullishScore > bearishScore) bullishScore += 2;
  else bearishScore += 2;
}

const netScore = bullishScore - bearishScore;
const strength = Math.abs(netScore) > 10 ? 'STRONG' :
                 Math.abs(netScore) > 5 ? 'MODERATE' : 'WEAK';

// Only alert if score differential > 3
if (Math.abs(netScore) > 3) {
  return {
    signal: netScore > 0 ? 'BULLISH' : 'BEARISH',
    strength: strength,
    bullishScore: bullishScore,
    bearishScore: bearishScore
  };
}
```

```javascript
for (const item of items) {
  const symbol = item.json.symbol;
  const prices = item.json.prices; // Array of OHLCV

  // Calculate indicators
  const rsi = calculateRSI(prices.map(p => p.close), 14);
  const macd = calculateMACD(prices.map(p => p.close));
  const bb = calculateBollingerBands(prices.map(p => p.close), 20, 2);
  const volume = prices[prices.length - 1].volume;
  const avgVolume = prices.slice(-20).reduce((sum, p) => sum + p.volume, 0) / 20;

  // Scoring logic
  let bullishScore = 0, bearishScore = 0;

  if (rsi < 30) bullishScore += 2;
  if (rsi > 70) bearishScore += 2;

  if (macd.macd > macd.signal) bullishScore += 2;
  else bearishScore += 2;

  if (macd.histogram > 0) bullishScore += 1;
  else bearishScore += 1;

  const currentPrice = prices[prices.length - 1].close;
  if (currentPrice < bb.lower) bullishScore += 2;
  if (currentPrice > bb.upper) bearishScore += 2;

  if (volume > avgVolume * 1.5) {
    if (bullishScore > bearishScore) bullishScore += 2;
    else bearishScore += 2;
  }

  const netScore = bullishScore - bearishScore;

  // Only output if signal strong enough
  if (Math.abs(netScore) > 3) {
```

```javascript
function validateTrade(trade) {
  const checks = [];

  // 1. Position size within 2% rule
  if (trade.maxRisk > account.balance * 0.02) {
    checks.push({
      passed: false,
      reason: "Max risk exceeds 2% rule",
      teaching: "The 2% rule protects your account from ruin. Even with 10 consecutive losses, you'd only be down 20%. This is Hermetic Principle #6: Cause & Effect."
    });
  } else {
    checks.push({ passed: true });
  }

  // 2. Portfolio correlation check
  const correlation = calculateCorrelation(trade.symbol, portfolio);
  if (correlation > 0.7) {
    checks.push({
      passed: false,
      reason: `High correlation (${correlation}) with existing position`,
      teaching: "Correlated positions = concentrated risk. If one fails, they often all fail together. Diversify across uncorrelated assets."
    });
  } else {
    checks.push({ passed: true });
  }

  // 3. Total portfolio risk check
  const totalRisk = calculateTotalPortfolioRisk(portfolio, trade);
  if (totalRisk > account.balance * 0.10) {
    checks.push({
      passed: false,
      reason: "Total portfolio risk would exceed 10%",
      teaching: "Never let aggregate risk exceed 10% of account. This ensures you can survive extended drawdowns."
    });
  } else {
    checks.push({ passed: true });
  }

  // 4. Market condition alignment
  if (trade.strategy === 'iron_condor' && vix > 25) {
    checks.push({
      passed: false,
      reason: "Iron condor in high volatility environment",
      teaching: "Iron condors profit from low volatility. High VIX means big moves likely. This is Principle #2: Correspondence - match strategy to environment."
    });
  } else {
    checks.push({ passed: true });
  }

  // 5. Mental clarity check (manual input required)
  if (!trade.thesisArticulated) {
    checks.push({
      passed: false,
      reason: "Trading thesis not clearly articulated",
      teaching: "Principle #1: Mentalism. If you can't write down WHY you're taking this trade, you don't understand it well enough."
    });
  } else {
    checks.push({ passed: true });
  }

  const allPassed = checks.every(c => c.passed);

  return {
    approved: allPassed,
    checks: checks,
    message: allPassed ?
      "Trade approved. All risk checks passed." :
      "Trade rejected. Fix issues before proceeding."
  };
}
```
